usage: fleet soak [-h] [--log-level {DEBUG,INFO,WARNING,ERROR}]
                  [--hosts H,H,…] [--discover-timeout SEC] [--board CLASS]
                  [--fields F,F,…] [--gate NAME] [--skip NAME]
                  [--out-json PATH] [--out-junit PATH] [--baseline PATH]
                  [--criteria PATH] [--settle [SEC]] [--dry-run] [--yes]
                  [--metrics-mqtt-url HOST[:PORT]] [--metrics-topic PREFIX]
                  [--no-publish-metrics] [--duration DURATION]
                  [--interval INTERVAL] [--target VERSION]
                  [--expected-ghs EXPECTED_GHS] [--quiet] [--samples-out PATH]
                  [--attach-logs {anomaly,always,never}]

options:
  -h, --help            show this help message and exit
  --log-level {DEBUG,INFO,WARNING,ERROR}
                        log level (default: WARNING)
  --duration DURATION   soak window: '30s', '2m', '1h', or bare seconds
                        (default: criteria.duration)
  --interval INTERVAL   poll interval in seconds (default:
                        criteria.poll_interval)
  --target VERSION      expected firmware version; fail devices not running
                        this version
  --expected-ghs EXPECTED_GHS
                        hashrate floor override in GH/s (default: read from
                        device /api/stats); 0 disables floor check
  --quiet, --no-progress
                        suppress per-tick live reporting (default: reporting
                        on)
  --samples-out PATH    write per-tick samples to file (.json = NDJSON, .csv
                        with header)
  --attach-logs {anomaly,always,never}
                        attach device log tail to result: anomaly (default),
                        always, never

targeting:
  --hosts H,H,…         comma-separated IPs/hostnames (skip mDNS discovery)
  --discover-timeout SEC
                        mDNS browse window in seconds (default: 10)
  --board CLASS         filter devices by board class substring (e.g. bitaxe)

output:
  --fields F,F,…        comma-separated field subset for on-demand validation
  --gate NAME           enable a specific check gate (repeatable; default:
                        all)
  --skip NAME           disable a specific check gate (repeatable)
  --out-json PATH       write JSON results to file
  --out-junit PATH      write JUnit XML results to file
  --baseline PATH       compare results against a prior JSON result file
  --criteria PATH       YAML criteria override file

run control:
  --settle [SEC]        opt-in warmup settle: bare = criteria default (120s),
                        --settle N = N seconds
  --dry-run             log mutating operations but don't execute them
  --yes                 auto-confirm guard for mutating operations

metrics publishing:
  --metrics-mqtt-url HOST[:PORT]
                        broker for run metrics (overrides
                        BB_TEST_METRICS_BROKER / BB_TEST_RECEIVER); default ON
                        when configured, skip with note when absent
  --metrics-topic PREFIX
                        MQTT topic prefix for run metrics (default:
                        fleettest); full topic: <prefix>/<suite>/<board>
  --no-publish-metrics  disable automatic metrics publishing (opt-out)
